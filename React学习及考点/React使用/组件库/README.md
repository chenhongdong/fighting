# 安装依赖
```
// react及react-dom相关
npm i react @types/react react-dom @types/react-dom -S

// webpack相关
npm i webpack webpack-cli webpack-dev-server html-webpack-plugin -D

// ts及loader相关
npm i typescript ts-loader source-map-loader style-loader css-loader less-loader less file-loader url-loader -D


// 单元测试相关
npm i jest @types/jest ts-jest jest-junit enzyme @types/enzyme -D


// 后端相关
npm i axios express qs @types/qs -D
```
